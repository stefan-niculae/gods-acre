# views.maintenance.fuzzy_year
import re
years_str = ' \'14, 2012 ,93 , \' 94  1969  '
years_list = list(map(int, re.findall('\d+', years_str)))
def fuzzy_year(yr):
    year_breakpoint = 50
    # 15 => 2015
    if yr <= year_breakpoint:
        return 2000 + yr
    # 94 => 1994
    if year_breakpoint < yr < 100:
        return 1900 + yr
    return yr
years_list = list(map(fuzzy_year, years_list))
